async function updateDashboard() { try { // Menambahkan angka acak di akhir link agar browser tidak pakai cache const cacheBuster = new Date().getTime(); const response = await fetch('data.json?t=' + cacheBuster); const data = await response.json(); document.getElementById('price').innerText = "$" + data.price; document.getElementById('rsi').innerText = data.rsi; document.getElementById('signal').innerText = data.signal; document.getElementById('update').innerText = "Last Update: " + data.last_update; const sigBox = document.getElementById('signal'); if (data.signal.includes('BUY')) { sigBox.className = "text-xl font-bold text-green-500 blink"; } else if (data.signal.includes('SELL')) { sigBox.className = "text-xl font-bold text-red-500 blink"; } else { sigBox.className = "text-xl font-bold text-yellow-500"; } } catch (e) { console.log("Updating data..."); } }
