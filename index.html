async function updateDashboard() { try { const response = await fetch('data.json?t=' + new Date().getTime()); const data = await response.json(); document.getElementById('price').innerText = "$" + data.price; // Kita tambahkan info RSI di sini document.getElementById('status').innerHTML = `
RSI: ${data.rsi}
Sinyal: ${data.signal}
`; document.getElementById('update').innerText = "Terakhir Update: " + data.last_update; } catch (e) { console.log("Menunggu data..."); } }
